Advanced Functions

var myArray = ['sun', 'moon', 'star'];

var myObj = {
    name: 'Matt',
    married: true,
    city: 'Austin'};


function combo(collection,callback ){
    if(collection.constructor === Array){
    
    for(i = 0; i < collection.length; i++){
        callback(collection[i]);
    }
    }else{for(x in collection){
        callback(collection[x]);
    	}
	}
   
};

combo(myArray,function(data){
console.log(data)
});
combo(myObj, function(data){
console.log(data)
});
